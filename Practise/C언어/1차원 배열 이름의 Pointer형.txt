#include <stdio.h>

int main(void) {

  int arr1[3] = {1, 2, 3};
  double arr2[3] = {1.1, 2.2, 3.3};

  //%d로 4바이트 정수형 출력, %g로 8바이트 실수형 출력
  printf("%d %g \n", *arr1, *arr2);

  //참조만할 수 있는것이 아니라 메모리 공간에 접근하여 값을 증가.
  *arr1 += 100;
  *arr2 += 120.5;

  printf("%d %g \n", arr1[0], arr2[0]);

  return 0;
}
